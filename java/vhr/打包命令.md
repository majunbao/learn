javac -d out src/com/example/*.java
jar cfe myapp.jar com.example.Main -C out .
java -jar myapp.jar

## 有依赖
# 1. 去官网下载 mysql-connector-java-8.0.33.jar
# 2. 放到 libs/ 目录
# 3. 编译时指定 classpath
javac -d out -cp "libs/*" src/com/example/*.java

# 4. 打包时需要把依赖也打进去（或者运行时手动 -cp）
jar cfe myapp.jar com.example.Main -C out .

# 5. 运行时还要带上依赖
java -cp myapp.jar:libs/mysql-connector-java-8.0.33.jar com.example.Main
